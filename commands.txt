# --- Setup Instructions ---

# 1. Clone repository
git clone https://github.com/YOUR_USERNAME/Gamora-GameBuilder.git
cd Gamora-GameBuilder

# 2. Create a virtual environment (optional but recommended)
python -m venv venv
# On Windows
venv\Scripts\activate
# On Mac/Linux
source venv/bin/activate

# 3. Install dependencies
pip install -r requirements.txt

# 4. Run the Streamlit app
streamlit run app.py

# 5. Generate & download games
# After download, run the game locally:
cd YourGameName
python main.py
